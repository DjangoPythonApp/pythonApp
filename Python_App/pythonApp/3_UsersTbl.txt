Users Table - 25 FAQ Queries:
1. Display All Records
SELECT * FROM users;

2. Display Specific Columns
SELECT username, email FROM users;

3. Find All Buyers
SELECT * FROM users WHERE role = 'Buyer';

4. Multiple Conditions
SELECT * FROM users WHERE role='Buyer' AND username='john';

5. OR Condition
SELECT * FROM users WHERE role='Buyer' OR role='Admin';

6. NOT Condition
SELECT * FROM users WHERE role <> 'Buyer';

7. IN Operator
SELECT * FROM users WHERE role IN ('Buyer','Seller');

8. LIKE Operator
SELECT * FROM users WHERE username LIKE 'j%';

9. Count Records
SELECT COUNT(*) FROM users;

10. Count Buyers
SELECT COUNT(*) FROM users WHERE role='Buyer';

11. DISTINCT
SELECT DISTINCT role FROM users;

12. ORDER BY ASC: ASC means descending (smallest to largest).
SELECT * FROM users ORDER BY username ASC;

13. ORDER BY DESC: DESC means descending (largest to smallest).
SELECT * FROM users ORDER BY username DESC;

14. LIMIT: First Two Users
SELECT * FROM users LIMIT 2;

15. OFFSET: Useful for pagination.
SELECT * FROM users LIMIT 2 OFFSET 1;

16. GROUP BY
SELECT role, COUNT(*) AS total_users FROM users GROUP BY role;

17. HAVING
SELECT role, COUNT(*) AS total_users FROM users GROUP BY role HAVING COUNT(*) > 1;

18. UPDATE
UPDATE users SET role='Admin' WHERE id=1;

19. DELETE
DELETE FROM users WHERE id=4;

20. Latest User: DESC means descending (largest to smallest).
SELECT * FROM users ORDER BY created_at DESC LIMIT 1;

21. Oldest User
SELECT * FROM users ORDER BY created_at ASC LIMIT 1;

22. Case-Insensitive Search
SELECT * FROM users WHERE username ILIKE '%JO%';

23. UPPER Function
SELECT username, UPPER(username) FROM users;

24. LENGTH Function
SELECT username, LENGTH(username) FROM users;

25. Combined Interview Query
SELECT role, COUNT(*) AS total_users FROM users GROUP BY role ORDER BY total_users DESC;
